SELECT m.member_name,r.review_text,DATE_FORMAT(r.review_date,'%Y-%m-%d')
FROM MEMBER_PROFILE m  INNER JOIN REST_REVIEW r
ON m.member_id=r.member_id
WHERE m.member_id = (SELECT member_id from rest_review group by member_id ORDER BY COUNT(*) DESC
    LIMIT 1)
ORDER BY r.review_date asc, review_text asc
